00349 // A structure that summarizes all the interesting attributes of a video or
00350 // audio track
00351 struct OmMediaSummary {
00352     enum OmMediaType type;           // as defined above
00353     enum OmVideoSampleRatio vsr;     // as defined above
00354     enum OmVideoAspectRatio aspect;  // as defined above
00355     uint bitrate;               // bitrate for the track; 0 if its embedded audio
00356     uint bitsPerUnit;           // 16 or 24 for audio (unit is a sample);
00357                                 // 8 or 10 for 601 video (unit is a pixel);
00358                                 // otherwise 0
00359     uint sampleRate;            // media sample rate * 100; e.g. NTSC video is 2997
00360                                 // 48khz audio is 4800000.
00361     uint channels;              // 1 for video; N for audio
00362 
00363     union Specific {
00364         struct Mpeg {
00365             uint gopLength;     // length of the gop structure; usually 15 for
00366                                 // MPEG video, but is one for everything else
00367             uint subGopLength;  // length of the sub-gop; usually 3 for IBBPBBP...
00368                                 // MPEG video; one for everything else
00369         } mpeg;
00370         struct Audio {
00371             enum OmAudioFormat format;
00372             unsigned char bigEndian;
00373             unsigned char sampleStride;
00374         } audio;
00375         struct Vbi {
00376             uint lineMask;
00377         } vbi;
00378     } specific;
